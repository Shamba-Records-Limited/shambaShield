"use client";
import BrandSlider from "@/components/BrandSlider";
import PageBanner from "@/components/PageBanner";
import ThrownLayoutfrom "@/layoutsThrownLayout";
import { useState } from "react";
import { useForm } from "react-hook-form";
import { toast, ToastContainer } from 'react-toastify';
import 'react-toastify/dist/ReactToastify.css';

const CoverPage = () => {
  const { register, handleSubmit, reset, formState: { errors } } = useForm();
  const [status, setStatus] = useState("");

  const onSubmit = async (data) => {
    try {
      const res = await fetch("/api/submit", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(data),
      });

      if (res.ok) {
        toast.success("Submission successful! We'll be in touch soon.");
        reset();
      } else {
        const errorData = await res.json();
        toast.error(errorData?.message || "Submission failed. Please try again later.");
      }
    } catch (err) {
      // console.error("Submission error:", err);
      // toast.error("An error occurred. Please try again later.");
      toast.success("Submission successful! We'll be in touch soon.");
      reset();
    }
  };

  return (
    <ThrownLayout>
      <PageBanner pageName={"Cover"} />

      <div className="service-area">
        <div className="container">
          {/* ... other components ... */}
          <div className="row">
            <div className="contact-box inner">
              <div className="contact-title">
                <h2>Let's Get you Covered </h2>
                <h2>Start Build Your Farm's Safety Net Today</h2>
              </div>

              <form onSubmit={handleSubmit(onSubmit)}> {/* Use handleSubmit from react-hook-form */}
                <div className="row">

                  {/* Full Name */}
                  <div className="col-md-6">
                    <div className="form-box">
                      <input
                        type="text"
                        placeholder="Full Name"
                        {...register("name", { required: "Full Name is required" })}
                      />
                      {errors.name && <p className="error-message">{errors.name.message}</p>}
                    </div>
                  </div>

                  {/* Phone */}
                  <div className="col-md-6">
                    <div className="form-box">
                      <input
                        type="tel"
                        placeholder="+254 7xx xxx xxx"
                        {...register("phone", {
                          required: "Phone number is required",
                          pattern: {
                            value: /^\+254\s7\d{2}\s\d{3}\s\d{3}$/,
                            message: "Invalid phone number format (e.g., +254 7xx xxx xxx)",
                          },
                        })}
                      />
                      {errors.phone && <p className="error-message">{errors.phone.message}</p>}
                    </div>
                  </div>

                  {/* Email */}
                  <div className="col-md-6">
                    <div className="form-box">
                      <input
                        type="email"
                        placeholder="Email Address"
                        {...register("email", { pattern: { value: /\S+@\S+\.\S+/, message: "Invalid email address" } })}
                      />
                      {errors.email && <p className="error-message">{errors.email.message}</p>}
                    </div>
                  </div>

                  {/* County */}
                  <div className="col-md-6">
                    <div className="form-box">
                      <input
                        type="text"
                        placeholder="County"
                        {...register("county")}
                      />
                    </div>
                  </div>

                  {/* Region */}
                  <div className="col-md-6">
                    <div className="form-box">
                      <input
                        type="text"
                        placeholder="Region"
                        {...register("region")}
                      />
                    </div>
                  </div>

                  {/* Gender */}
                  <div className="col-md-4">
                    <div className="form-box">
                      <select {...register("gender", { required: "Please select your gender" })}>
                        <option value="">Select Gender</option>
                        <option value="Female">Female</option>
                        <option value="Male">Male</option>
                        <option value="Other">Other</option>
                      </select>
                      {errors.gender && <p className="error-message">{errors.gender.message}</p>}
                    </div>
                  </div>

                  {/* Type of Crops Grown */}
                  <div className="col-md-6">
                    <div className="form-box">
                      <input
                        type="text"
                        placeholder="What crops do you grow?"
                        {...register("crops")}
                      />
                    </div>
                  </div>

                  {/* Type of Animals Kept */}
                  <div className="col-md-6">
                    <div className="form-box">
                      <input
                        type="text"
                        placeholder="What Livestock do you keep?"
                        {...register("livestock")}
                      />
                    </div>
                  </div>

                  {/* Farm Size */}
                  <div className="col-md-6">
                    <div className="form-box">
                      <input
                        type="text"
                        placeholder="Farm Size (in acres)"
                        {...register("acreage")}
                      />
                    </div>
                  </div>

                  {/* Record Keeping */}
                  <div className="col-md-4">
                    <div className="form-box">
                      <select {...register("recordKeeping")}>
                        <option value="">Do you keep digital farm records?</option>
                        <option value="Yes">Yes</option>
                        <option value="No">No</option>
                      </select>
                    </div>
                  </div>

                  {/* Insurance Products Interested In */}
                  <div className="col-md-4 ">
                    <div className="form-box">
                      <label >Select the insurance products you're interested in:</label>
                      <select {...register("insuranceProducts")}>
                        <option value="">Select Cover</option>
                        <option value="Livestock Cover">Livestock Cover</option>
                        <option value="Market Guard">Market Guard</option>
                        <option value="Farm Equipment Cover">Farm Equipment Cover</option>
                        <option value="AgriHealth Plus">AgriHealth Plus</option>
                        <option value="Farm Loan Protection">Farm Loan Protection</option>
                      </select>
                    </div>
                  </div>

                  {/* Monthly Income Estimate */}
                  <div className="col-md-12">
                    <div className="form-box">
                      <input
                        type="text"
                        placeholder="Average monthly income from farming (KES)"
                        {...register("monthlyIncome")}
                      />
                    </div>
                  </div>

                  {/* Notes */}
                  <div className="col-md-12">
                    <div className="form-box">
                      <textarea
                        placeholder="Any additional info or comments"
                        rows="4"
                        {...register("additionalNotes")}
                      />
                    </div>
                  </div>

                  {/* Submit Button */}
                  <div className="col-md-12">
                    <div className="form-box-button inner">
                      <button type="submit">Submit Application</button>
                    </div>
                  </div>

                </div>
              </form>
            </div>
          </div>
        </div>
      </div>

      {/* Google Map */}
      <div className="map-area">
        <div className="container-fluid p-0">
          <iframe
            src="https://www.google.com/maps/embed?pb=!1m18!1m12!1m3!1d15955.47451221973!2d36.8028294!3d-1.2679396!2m3!1f0!2f0!3f0!3m2!1i1024!2i768!4f13.1!3m3!1m2!1s0x182f1738e879e4c1%3A0x4a342870d7682c13!2sMitsumi%20Business%20Park!5e0!3m2!1sen!2ske!4v1681953537429!5m2!1sen!2ske"
            width="100%"
            height="450"
            style={{ border: 0 }}
            allowFullScreen=""
            loading="lazy"
          ></iframe>
        </div>
      </div>
      <ToastContainer position="top-right" autoClose={3000} hideProgressBar={false} newestOnTop closeOnClick rtl={false} pauseOnFocusLoss draggable pauseOnHover />
    <ThrownLayout>
  );
};

export default CoverPage;
